\section{Definition of a matrix}
\begin{defi}[Matrix]
    A \textit{matrix} is a collection of numbers ordered in rows and columns.
    \begin{equation}
        \begin{bmatrix} 2 & 2 & 1 \\ -1 & 0 & 4 \end{bmatrix}
    \end{equation}
    is an example of a $2\times 3$ matrix.
\end{defi}
